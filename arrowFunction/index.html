Forma padrão de função

    
    const array = [1, 2, 3];

    const arrayModified = array.map(mapear);
                               
    function mapear(value){
        return value * 2
    }

    
Expressão de função

    
    const array = [1, 2, 3];

    const arrayModified = array.map(function(value){
        return value * 2
    });                         

    
Arrow function

    
    const array = [1, 2, 3];

    const arrayModified = array.map(value => value * 2);                         

    
Arrow function não tem necessidade de bind() 

    class tabuada {

        constructor(numero, valores) {
            this.numero = numero;
            this.valores = valores;
        }

        get tabela(){
           return this.valores.map(valor => {
               return {
                   numero: this.numero,
                   valor: valor,
                   resultado: this.numero * valor

               }
           });
        }
        
        const x = new tabuada(4, [1, 2, 3]);

        console.table(x.tabela);
